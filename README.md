\begin{itemize}
    \item \textbf{Adjusted Stock Price \( S_* \)}:
    \begin{equation}
    S_* = S_0 - D
    \end{equation}

    \item \textbf{Maximum Profit \( P_{\text{max}} \)}:
    \begin{equation}
    P_{\text{max}} = S_0 - D
    \end{equation}

    \item \textbf{Maximum Loss \( L_{\text{max}} \)}:
    \begin{equation}
    L_{\text{max}} = K - S_0 + D
    \end{equation}
\end{itemize}
